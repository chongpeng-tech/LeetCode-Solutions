void reverseString(char* s, int sSize) {
    if(sSize%2==0){
        for(int i = 0;i<sSize/2;i++){
            char temp = s[i];
            s[i] = s[sSize-i-1];
            s[sSize-i-1] = temp;
        }
    }else{
        for(int i = 0;i<(sSize-1)/2;i++){
            char temp = s[i];
            s[i] = s[sSize-i-1];
            s[sSize-i-1] = temp;
        }
    }
    return;    
}
