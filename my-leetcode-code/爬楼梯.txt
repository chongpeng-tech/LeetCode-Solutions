//单纯递归超时，效率太低
//记忆化递归，减少重复计算
int climb(int n, int* store){
    if(store[n] != -1){
        return store[n];
    }else{
        store[n] = climb(n - 1, store) + climb(n - 2, store);
        return store[n];
    }
}
int climbStairs(int n) {
    int* store = (int*)malloc((n+1)*sizeof(int));
    store[0] = 1;
    store[1] = 2;
    for(int i = 2; i < n; i++){
        store[i] = -1;
    }
    int res = climb(n - 1 , store);
    free(store);
    return res;
}
