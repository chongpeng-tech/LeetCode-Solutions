int removeElement(int* nums, int numsSize, int val) {
    int returnSize;
    returnSize = numsSize;
    for(int i = 0; i < returnSize; i++){
        if(nums[i] == val){
            for(int j = i; j < returnSize - 1; j++){
                nums[j] = nums[j + 1];
            }
            returnSize--;
            i--;
        }
    }
    return returnSize;
}
