/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     struct ListNode *next;
 * };
 */
//反转链表
struct ListNode* reverseList(struct ListNode* head) {
    if(head==NULL||head->next==NULL){
        return head;
    }
    struct ListNode* prev = NULL;
    struct ListNode* currentNode = head;
    struct ListNode* nextNode = head->next;
    while(1){
        currentNode->next = prev;
        prev = currentNode;
        if(nextNode==NULL){
            break;//什么时候终止很重要（快排？）
        }
        currentNode = nextNode;
        nextNode = nextNode->next;

    }
    return currentNode;
}
//找中间节点
struct ListNode* middleNode(struct ListNode* head) {
    int len_list = 0;
    struct ListNode* currerntNode = head;
    if(head->next==NULL){
        return head;
    }
    //先找出链表的长度
    while(currerntNode!=NULL){
        currerntNode = currerntNode->next;
        len_list++;
    }
    if(len_list%2!=0){
        //基数个节点，返回中间的节点
        currerntNode = head;
        for(int i = 0;i<len_list/2;i++){
            currerntNode = currerntNode->next;
        }
        return currerntNode;
    }else{
        //偶数个节点返回第二个中间节点
        currerntNode = head;
        for(int i = 0;i<len_list/2;i++){
            currerntNode = currerntNode->next;
        }
        return currerntNode;
    }

}
