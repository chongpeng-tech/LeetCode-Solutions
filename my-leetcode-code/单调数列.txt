bool isMonotonic(int* nums, int numsSize) {
    int sign = nums[numsSize-1]-nums[0];

    for(int i = 0;i<numsSize-1;i++){
        if(nums[i]>nums[i+1]&&sign>0){
            return false;
        }
        else if(nums[i]<nums[i+1]&&sign<0){
            return false;
        }else if(nums[i]-nums[i+1]!=0&&sign==0){
            return false;
        }
        
            

    }
    return true;
}
