int arraySign(int* nums, int numsSize) {
    int product = 1;
    int temp;
    for(int i = 0;i<numsSize;i++){
        temp = nums[i];
        product = product * temp;
        if(product>100||product<-100){
            product = product / 100;
        }
    }
    int signFunc(int x){
        if(x>0){
            return 1;
        }else if(x<0){
            return -1;
        }else{
            return 0;
        }
    }
    return signFunc(product);
}
